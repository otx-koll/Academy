#### 7. 하루의 할 일을 표현하는 클래스 Day는 다음과 같다. 한 달의 할 일을 표현하는 MonthSchedule 클래스 작성하라.
Day클래스
```
public class Day {
	private String work;
	public void set(String work) {
		this.work = work;
	}
	public String get() {
		return work;
	}
	public void show() {
		if(work == null)
			System.out.println("없습니다.");
		else
			System.out.println(work+"입니다.");
	}
}
```

MonthSchedule 클래스
```
import java.util.Scanner;

public class MonthSchedule {
	Scanner s = new Scanner(System.in);
	
	int dayNum;
	Day day[];
	boolean state = true;
	
	public MonthSchedule(int a) {
		dayNum = a;
		day = new Day[dayNum];
		for(int i=0; i<day.length; i++) {
			day[i] = new Day();
		}
	}
	private void input() {
		System.out.print("날짜(1~30)?");
		int date = s.nextInt();
		System.out.print("할일(빈칸없이 입력)?");
		String work = s.next();
		day[date-1].set(work);
	}
	private void view() {
		System.out.print("날짜(1~30)?");
		int date = s.nextInt();
		System.out.print(date+"일의 할 일은 ");
		day[date-1].show();
	}
	private void finish() {
		System.out.println("프로그램을 종료합니다.");
		state = false;
	}
	public void run() {
		System.out.println("이번달 스케쥴 관리 프로그램");
		while(state) {
			System.out.print("할 일(입력:1, 보기:2, 끝내기:3) >>");
			int opinion = s.nextInt();
			switch(opinion) {
			case 1:
				input();
				break;
			case 2:
				view();
				break;
			case 3:
				finish();
				break;
			default:
				System.out.println("잘못입력하였습니다.");
			}
		}
		System.out.println();
	}
}
```
Main 클래스
```
public class Main {

	public static void main(String[] args) {
		new MonthSchedule(30).run();
	}
}
```
